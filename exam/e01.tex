\newcommand{\answerbox}[3]{ % usage: \answerbox{Text:}{width}{height}
    \addvbuffer[12pt 8pt]{ % adds vertical/horizontal space on top of box
    \begin{tabular}{|l p{#2}|}
        \hline
            #1 & \\ [#3]
        \hline
    \end{tabular}
    } 
}
\newcommand{\codelinebox}[1]{ % usage eg: \codelinebox{x =}
    \addvbuffer[0pt]{ % adds vertical/horizontal space on top of box
    \hspace{1em}
    \begin{tabular}{|l p{5in}|}
        \hline
                     & \\ [.10in]
            {\tt #1} & \\ [.25in]
        \hline
    \end{tabular}
    }
}
